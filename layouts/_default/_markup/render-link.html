{{- /* Link render hook. A bare-URL autolink -- ``, as used by the citations in sidenotes -- is rendered as just its hostname followed by a small link glyph, while the anchor still resolves to the full URL. Ordinary labelled links, `[text](url)`, are left to render normally. */ -}} {{- $dest := .Destination -}} {{- $u := urls.Parse $dest -}} {{- $text := .Text | plainify -}} {{- if and (eq $text $dest) $u.Host -}} {{- $host := strings.TrimPrefix "www." $u.Host -}} {{ $host }}↗ {{- else -}} {{ .Text }} {{- end -}}
